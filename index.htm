                return;

            const adminPassword = adminPasswordInput.value;

                showMessage("Por favor, introduce el correo y la contraseña del administrador.", true);

                return;

            }

 

            showMessage("Verificando credenciales de administrador...", false);

            try {

                const response = await fetch(appsScriptUrl, {

                    method: 'POST',

                    mode: 'no-cors', // Importante para cross-origin requests

                    headers: {

                        'Content-Type': 'application/json',

                    },

                    body: JSON.stringify({

                        action: 'adminLogin', // Indicar al Apps Script que es una solicitud de login de admin

                        email: adminEmail,

                        password: adminPassword

                    }),

                });

 

                // No-cors mode means we can't directly read response.json(),

                // but the request will be sent. For verification, we assume success

                // if there's no network error, and rely on Apps Script logs for debugging.

                // In a real app, you'd configure CORS to read the JSON response.

 

                // For this example, we'll assume success if the fetch doesn't throw.

                // A more robust client-side check would involve a response from Apps Script

                // that client can read (requires CORS config on Apps Script side).

                // Given "no-cors", we can't reliably read the `result.success` from here.

                // So, for demonstration, we'll *assume* success if the request was sent without network error.

                // If the user reports it's not working after this, then we need to enable CORS for proper error handling.

 

                // For now, after successful request send, we rely on the internal Apps Script validation.

                // This means the success message here is optimistic.

                showMessage("Credenciales enviadas para verificación. Accediendo...", false);

                setAppMode('adminView');

                adminEmailInput.value = '';

                adminPasswordInput.value = '';

                showMessage(`Error de conexión al intentar acceder como administrador. Inténtalo de nuevo.`, true);

            }

                isAdminActionLoading = false;

                adminLoadingMessage.classList.add('hidden');

                renderAdminClients();

                updateAdminButtonState();

            }

        };

 

        const handleClientSelection = (clientId, isChecked) => {

            if (isChecked) {

                selectedClientIds.push(clientId);

            } else {

                selectedClientIds = selectedClientIds.filter(id => id !== clientId);

            }

            updateAdminButtonState();

        };

 

        // Marks selected clients as logically deleted (CONDICION: false) and updates Google Sheet

        const handleMarkClientsAsDeleted = async () => {

            if (selectedClientIds.length === 0) {

                showMessage("Selecciona al menos un cliente para marcar como borrado.", true);

                return;

            }

 

            isAdminActionLoading = true;

            showMessage("Marcando clientes seleccionados como borrados...");

            updateAdminButtonState();

 

            try {

                const clientIdsToMarkDeleted = [...selectedClientIds];

               

                selectedClientIds = [];

                updateAdminButtonState();

 

                const updatePromises = clientIdsToMarkDeleted.map(clientId => {

                    const clientToUpdate = adminClients.find(c => c.Cliente_IDX === clientId);

 

                    if (clientToUpdate) {

                        return fetch(appsScriptUrl, {

                            method: 'POST',

                            mode: 'no-cors',

                            headers: {

                                'Content-Type': 'application/json',

                            },

                            body: JSON.stringify({

                                id: clientToUpdate.Cliente_IDX,

                                CONDICION: false

                            }),

                        });

                    }

                    return Promise.resolve();

                });

                await Promise.all(updatePromises);

 

                await fetchAllClientsForAdmin();

                showMessage("Clientes seleccionados marcados como borrados en Google Sheet y lista actualizada.");

               

            } catch (error) {

                console.error("Error marking selected clients as deleted:", error);

                showMessage(`Error al marcar clientes como borrados en Google Sheet: ${error.message}`, true);

            } finally {

                isAdminActionLoading = false;

                updateAdminButtonState();

            }

        };

 

        const handleAdminLogout = () => {

            adminClients = [];

            selectedClientIds = [];

            setAppMode('auth');

            showMessage("Sesión de administrador cerrada.");

        };

 

        // --- Initial Setup and Event Listeners ---

        window.onload = () => {

            setAppMode('auth');

            userLoginForm.addEventListener('submit', handleLogin);

            adminLoginForm.addEventListener('submit', handleAdminLogin);

            userLogoutBtn.addEventListener('click', handleLogout);

            markDeletedBtn.addEventListener('click', handleMarkClientsAsDeleted);

            adminLogoutBtn.addEventListener('click', handleAdminLogout);

        };

    </script>
